You need to define selectedCount for the "spreads" category at a higher scope, so it's accessible both when you're iterating through the individual spread items (to determine isAtLimit) and when you're deciding whether to show the serving size selection.

Here's how you can adjust the code within the AppetizersStep component, specifically in the if (category.id === "spreads") block:

// Inside AppetizersStep component, within horsDoeurvesData.categories.map(...)

if (category.id === "spreads") {
  // Calculate selectedCount for the current "spreads" category ONCE here
  const currentSpreadsCategorySelections = horsDoeurvesSelections?.categories?.[category.id]?.items;
  const selectedCountForThisSpreadsCategory = currentSpreadsCategorySelections ? Object.keys(currentSpreadsCategorySelections).length : 0;

        )}
      </div>

      <div className="grid grid-cols-1 md:grid-cols-2 gap-3 mb-4">

                        handleHorsDoeurvesItemSelection(category.id, item.id, category.servingSizes?.[0] || 24);

                      handleHorsDoeurvesItemSelection(category.id, item.id, null);
                    }
                  }}
                  disabled={isAtLimit}
                  className="mr-3"
                />
                <div className="font-medium">{item.name}</div>
              </label>
            </div>
          );

      </div>

                  handleHorsDoeurvesItemSelection(category.id, itemId, quantity);

          >
            {category.servingSizes.map((size) => (

            ))}
          </RadioGroup>
        </div>
      )}
    </div>
  );
}
