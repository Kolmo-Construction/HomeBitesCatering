Add Priority Column to columns Definition:

// Within the 'columns: ColumnDef<Opportunity>[]' array

{
  accessorKey: "priority",
  header: "Priority",
  cell: ({ row }) => {
    const priority = row.original.priority; // Assuming 'priority' is a field on your Opportunity type
    let badgeClass = "";
    let textColor = "text-white"; // Default text color for badges

    switch (priority) {
      case 'hot':
        badgeClass = "bg-red-600 hover:bg-red-700"; // More intense red for hot
        break;
      case 'high':
        badgeClass = "bg-orange-500 hover:bg-orange-600";
        break;
      case 'medium':
        badgeClass = "bg-yellow-500 hover:bg-yellow-600";
        textColor = "text-gray-800"; // Darker text for yellow bg
        break;
      case 'low':
        badgeClass = "bg-blue-500 hover:bg-blue-600";
        break;
      default:
        badgeClass = "bg-gray-400 hover:bg-gray-500";
    }
    // The `BadgeStatus.txt` component is for predefined status strings.
    // For priority, a direct `Badge` is fine, or you could adapt `BadgeStatus`
    // if you want more complex label/color configurations managed centrally.
    return (
      <Badge className={cn("capitalize px-2.5 py-1 text-xs font-semibold", badgeClass, textColor)}>
        {priority || "N/A"}
      </Badge>
    );
  },
  // Optional: Enable sorting for this column
  enableSorting: true,
},
