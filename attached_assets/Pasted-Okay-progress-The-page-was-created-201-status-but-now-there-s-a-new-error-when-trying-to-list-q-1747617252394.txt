Make mergeMetadata (and mergeOptions) Robust Against null Inputs:
Modify your helper functions in formBuilderRoutes.ts:

TypeScript

// clea/server/formBuilderRoutes.ts

function mergeMetadata(baseMetadata, overrides) {
  const B = baseMetadata || {}; // Default to empty object if null/undefined
  const O = overrides || {};   // Default to empty object if null/undefined

  // If both are empty or effectively empty after defaulting, return an empty object or one of them
  if (Object.keys(B).length === 0 && Object.keys(O).length === 0) return {};
  if (Object.keys(B).length === 0) return O;
  if (Object.keys(O).length === 0) return B;

  const result = { ...B }; // Now B is guaranteed to be an object

  for (const key in O) {
    if (Object.prototype.hasOwnProperty.call(O, key)) { // Ensure it's an own property
      if (
        typeof O[key] === 'object' && O[key] !== null && !Array.isArray(O[key]) &&
        typeof result[key] === 'object' && result[key] !== null && !Array.isArray(result[key])
      ) {
        result[key] = mergeMetadata(result[key], O[key]);
      } else if (Array.isArray(O[key])) {
        result[key] = [...O[key]];
      } else {
        result[key] = O[key];
      }
    }
  }
  return result;
}

function mergeOptions(baseOptions, overrides) {
  // If baseOptions is an array (typical for select/radio/checkbox)
  if (Array.isArray(baseOptions)) {
    if (Array.isArray(overrides) && overrides.length > 0) { // If overrides is a non-empty array, it takes precedence
      return [...overrides];
    }
    // Add more sophisticated array merging if needed, like by item 'key' or 'value'
    return baseOptions; // Default to base if overrides are not a replacing array
  }

  // If options are an object (e.g., for matrix structure within optionsOverrides)
  // Use the robust mergeMetadata logic
  const B = baseOptions || {};
  const O = overrides || {};

  if (Object.keys(B).length === 0 && Object.keys(O).length === 0) return {};
  if (Object.keys(B).length === 0) return O;
  if (Object.keys(O).length === 0) return B;

  return mergeMetadata(B, O); // Reuse robust object merging
}
