router.delete("/questions/:questionId/rules", async (req, res) => {
  try {
    const questionInstanceId = parseInt(req.params.questionId);

    if (isNaN(questionInstanceId)) {
      return res.status(400).json({ message: "Invalid question ID" });
    }
    console.log(`SERVER: Deleting rules targeting question instance ID: ${questionInstanceId}`);

    // 1. Find all rule IDs where this question instance is a target.
    const targetEntries = await db
      .select({ ruleId: formSchema.formRuleTargets.ruleId })
      .from(formSchema.formRuleTargets)
      .where(
        and(
          eq(formSchema.formRuleTargets.targetId, questionInstanceId),
          eq(formSchema.formRuleTargets.targetType, "question")
        )
      );

    const ruleIdsToDelete = targetEntries.map((t) => t.ruleId);
    console.log(`SERVER: Found ${ruleIdsToDelete.length} rules targeting this question.`);

    if (ruleIdsToDelete.length > 0) {
      // 2. Delete these specific target entries from form_rule_targets.
      const deleteTargetsResult = await db
        .delete(formSchema.formRuleTargets)
        .where(
          and(
            inArray(formSchema.formRuleTargets.ruleId, ruleIdsToDelete),
            eq(formSchema.formRuleTargets.targetId, questionInstanceId),
            eq(formSchema.formRuleTargets.targetType, "question")
          )
        ).returning(); // Add returning to see what was deleted
      console.log(`SERVER: Deleted ${deleteTargetsResult.length} target entries.`);


      // 3. For each rule that was affected, check if it has any other targets.
      // If a rule has no targets left, delete the rule itself from form_rules.
      let orphanedRulesDeletedCount = 0;
      for (const ruleId of ruleIdsToDelete) {
        const remainingTargets = await db
          .select({ count: sql<number>`count(*)` })
          .from(formSchema.formRuleTargets)
          .where(eq(formSchema.formRuleTargets.ruleId, ruleId));

        if (Number(remainingTargets[0].count) === 0) {

          await db
            .delete(formSchema.formRules)

        } else {
          console.log(`SERVER: Rule ID ${ruleId} still has ${remainingTargets[0].count} targets, not deleting rule itself.`);
        }
      }
      console.log(`SERVER: Deleted ${orphanedRulesDeletedCount} orphaned rules.`);
    } else {
        console.log(`SERVER: No rules found targeting question ${questionInstanceId}. Nothing to delete.`);
