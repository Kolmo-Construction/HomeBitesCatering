// server/questionLibraryRoutes.ts
async function getQuestionWithDetails(id: number) {
  // Get the base question
  const questionResult = await db.execute(sql`SELECT * FROM question_library WHERE id = ${id}`);

  if (questionResult.rows.length === 0) {
    return null;
  }
  const question = questionResult.rows[0];

  // If it's a matrix question, get rows and columns
  if (question.question_type === 'matrix') { // Accessing property from raw SQL result
    const rowsResult = await db.execute(sql`SELECT * FROM library_matrix_rows WHERE library_question_id = ${id} ORDER BY row_order`);
    
    const columnsResult = await db.execute(sql`SELECT * FROM library_matrix_columns WHERE library_question_id = ${id} ORDER BY column_order`);
    
    return {
      ...question,
      matrixRows: rowsResult.rows,
      matrixColumns: columnsResult.rows
    };
  }
  
  return question;
}
Reasoning:

By using sql\SELECT * FROM question_library WHERE id = ${id}`, you are leveraging Drizzle'ssqltemplate tag. This allows Drizzle to properly process the query and its parameters, ensuring that${id}` is correctly substituted, thus resolving the "no parameter $1" error. This is the recommended way to execute raw, parameterized SQL queries with Drizzle ORM to ensure security and correctness.
