Example Fetch Logic (Conceptual):

useEffect(() => {
  if (formId && formData) { // Assuming formData is loaded and contains the form structure or you fetch separately
    const extractedQuestions: Array<{ id: string; displayText: string; pageId: string; questionKey: string }> = [];
    const extractedPages: Array<{ id: string; pageTitle: string }> = [];

    // Iterate through formData.pages (or pagesData if fetched separately)
    (formData.pages || pagesData || []).forEach(page => {
      extractedPages.push({ id: String(page.id || page.pageId), pageTitle: page.pageTitle || page.title || `Page ${page.pageOrder}` });
      (page.questions || []).forEach(questionInstance => {
        extractedQuestions.push({
          id: String(questionInstance.id || questionInstance.pageQuestionId),
          displayText: questionInstance.displayTextOverride || questionInstance.displayText || questionInstance.display_text || questionInstance.libraryDefaultText || 'Unnamed Question',
          pageId: String(page.id || page.pageId),
          questionKey: questionInstance.questionKey || `<span class="math-inline">\{page\.id \|\| page\.pageId\}\_</span>{questionInstance.id || questionInstance.pageQuestionId}`
        });
      });
    });
    setAllQuestionsInForm(extractedQuestions);
    setAllPagesInForm(extractedPages);
  }
}, [formId, formData, pagesData]); // Adjust dependencies as needed
This allQuestionsInForm and allPagesInForm state will be passed down to the QuestionSettingsPanel and then to the RuleEditorForm dialog.

// Inside QuestionSettingsPanel
const { data: conditionalRules = [], refetch: refetchConditionalRules } = useQuery<RuleDefinition[]>({
  queryKey: ['formQuestionRules', question?.id], // question.id is formPageQuestions.id
  queryFn: async () => {
    if (!question?.id) return [];
    const response = await fetch(`/api/form-builder/form-page-questions/${question.id}/rules`);
    if (!response.ok) {
      throw new Error('Failed to fetch conditional rules');
    }
    return response.json();
  },
  enabled: !!question?.id, // Only fetch if a question is selected
});

const saveRuleMutation = useMutation({
  mutationFn: async (ruleDataWithTargets: Omit<RuleDefinition, 'id'> & { id?: string }) => {
    const { id: ruleId, ...payload } = ruleDataWithTargets;
    const completePayload = {
      ...payload,
      formId: parseInt(formId), // Ensure formId is available and parsed
      triggerFormPageQuestionId: parseInt(question.id), // The question this panel is for
      // Ensure targets are correctly formatted with numeric IDs
      targets: payload.targets.map(t => ({
        ...t,
        targetId: parseInt(t.targetId, 10)
      }))
    };

    if (ruleId && !ruleId.startsWith('temp-')) { // Editing an existing rule (that has a real ID)
      const response = await fetch(`/api/form-builder/rules/${ruleId}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(completePayload),
      });
      if (!response.ok) throw new Error(await response.text());
      return response.json();
    } else { // Creating a new rule
      const response = await fetch(`/api/form-builder/forms/${formId}/rules`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(completePayload),
      });
      if (!response.ok) throw new Error(await response.text());
      return response.json();
    }
  },
  onSuccess: () => {
    toast({ title: "Rule saved", description: "Conditional rule has been successfully saved to the server." });
    refetchConditionalRules(); // Refetch rules for the current question
    setIsRuleEditorOpen(false);
    setCurrentRule(null);
  },
  onError: (error: Error) => {
    toast({ variant: "destructive", title: "Error saving rule", description: error.message });
  },
});

Deleting Rules (using useMutation):

Modify handleDeleteRule to use a mutation.
TypeScript

const deleteRuleMutation = useMutation({
  mutationFn: async (ruleId: string) => {
    if (ruleId.startsWith('temp-')) { // For locally added, unsaved rules
      return Promise.resolve({ message: "Local rule removed" });
    }
    const response = await fetch(`/api/form-builder/rules/${ruleId}`, {
      method: 'DELETE',
    });
    if (!response.ok) throw new Error(await response.text());
    return response.json();
  },
  onSuccess: (data, ruleId) => {
    toast({ title: "Rule deleted", description: "Conditional rule has been deleted." });
    if (ruleId.startsWith('temp-')) {
      // If it was a local rule, just filter it out from local state (already handled by setConditionalRules)
      // This part is a bit redundant if you're calling refetchConditionalRules anyway
      // but good for immediate UI update if not refetching immediately.
      // setConditionalRules(prevRules => prevRules.filter(r => r.id !== ruleId));
    }
    refetchConditionalRules(); // Refetch rules
  },
  onError: (error: Error) => {
    toast({ variant: "destructive", title: "Error deleting rule", description: error.message });
  },
});

const handleDeleteRule = (ruleId: string) => {
  // Optional: Add a confirmation dialog here
  if (window.confirm("Are you sure you want to delete this rule?")) {
    deleteRuleMutation.mutate(ruleId);
  }
};
