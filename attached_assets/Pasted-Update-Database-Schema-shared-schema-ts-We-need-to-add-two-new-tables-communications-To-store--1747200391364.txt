Update Database Schema (shared/schema.ts)

We need to add two new tables:

communications: To store all interaction records (emails, calls, notes).
contactIdentifiers: To manage multiple email addresses and phone numbers for leads and clients.
And we'll add a userId to communications to track who manually created an entry.

Here's the updated shared/schema.ts:

TypeScript

// shared/schema.ts
import { pgTable, text, serial, integer, boolean, timestamp, jsonb, doublePrecision, pgEnum } from "drizzle-orm/pg-core";
import { createInsertSchema } from "drizzle-zod";
import { z } from "zod";

// User and authentication tables
export const users = pgTable("users", {
  id: serial("id").primaryKey(),
  username: text("username").notNull().unique(),
  password: text("password").notNull(),
  firstName: text("first_name").notNull(),
  lastName: text("last_name").notNull(),
  email: text("email").notNull().unique(),
  phone: text("phone"),
  role: text("role").default("staff").notNull(), // admin, staff, kitchen, client
  createdAt: timestamp("created_at").defaultNow().notNull(),
});

export const insertUserSchema = createInsertSchema(users).omit({
  id: true,
  createdAt: true
});

// Leads
export const leads = pgTable("leads", {
  id: serial("id").primaryKey(),
  firstName: text("first_name").notNull(),
  lastName: text("last_name").notNull(),
  email: text("email").notNull(), // This can be the primary/initial email
  phone: text("phone"), // This can be the primary/initial phone
  eventType: text("event_type").notNull(),
  eventDate: timestamp("event_date"),
  guestCount: integer("guest_count"),
  venue: text("venue"),
  notes: text("notes"), // General notes about the lead
  status: text("status").default("new").notNull(),
  leadSource: text("lead_source"),
  assignedTo: integer("assigned_to").references(() => users.id),
  clientId: integer("client_id").references(() => clients.id), // Link to client if converted
  createdAt: timestamp("created_at").defaultNow().notNull(),
  updatedAt: timestamp("updated_at").defaultNow().notNull(),
});

export const insertLeadSchema = createInsertSchema(leads, {
  eventDate: z.string().nullable().transform(date => date ? new Date(date) : null),
}).omit({
  id: true,
  createdAt: true,
  updatedAt: true,
  // clientId is handled by lead conversion process or specific assignment
});

// Clients
export const clients = pgTable("clients", {
  id: serial("id").primaryKey(),
  userId: integer("user_id").references(() => users.id), // If client has a user account
  firstName: text("first_name").notNull(),
  lastName: text("last_name").notNull(),
  email: text("email").notNull(), // Primary email for the client
  phone: text("phone"), // Primary phone for the client
  company: text("company"),
  address: text("address"),
  city: text("city"),
  state: text("state"),
  zip: text("zip"),
  notes: text("notes"), // General notes about the client
  leadId: integer("lead_id").references(() => leads.id), // Original lead ID if converted
  createdAt: timestamp("created_at").defaultNow().notNull(),
  updatedAt: timestamp("updated_at").defaultNow().notNull(),
});

export const insertClientSchema = createInsertSchema(clients).omit({
  id: true,
  createdAt: true,
  updatedAt: true
});

// --- NEW: Contact Identifiers Table ---
export const identifierTypeEnum = pgEnum("identifier_type", ["email", "phone"]);

export const contactIdentifiers = pgTable("contact_identifiers", {
  id: serial("id").primaryKey(),
  leadId: integer("lead_id").references(() => leads.id, { onDelete: 'cascade' }), // Link to lead
  clientId: integer("client_id").references(() => clients.id, { onDelete: 'cascade' }), // Link to client
  type: identifierTypeEnum("type").notNull(), // 'email' or 'phone'
  value: text("value").notNull(), // The actual email address or phone number
  isPrimary: boolean("is_primary").default(false).notNull(),
  source: text("source"), // How this identifier was added (e.g., 'lead_form', 'email_sync', 'manual_entry')
  createdAt: timestamp("created_at").defaultNow().notNull(),
  // Optional: Add unique constraint for (leadId, type, value) and (clientId, type, value) if needed
  // Though, it might be better to handle duplicates at the application layer when adding.
});

export const insertContactIdentifierSchema = createInsertSchema(contactIdentifiers).omit({
  id: true,
  createdAt: true,
});


// --- NEW: Communications/Interactions Table ---
export const communicationTypeEnum = pgEnum("communication_type", ["email", "call", "sms", "note", "meeting"]);
export const communicationDirectionEnum = pgEnum("communication_direction", ["incoming", "outgoing", "internal"]);

export const communications = pgTable("communications", {
  id: serial("id").primaryKey(),
  leadId: integer("lead_id").references(() => leads.id, { onDelete: 'set null' }), // Link to lead
  clientId: integer("client_id").references(() => clients.id, { onDelete: 'set null' }), // Link to client
  userId: integer("user_id").references(() => users.id, { onDelete: 'set null' }), // User who created/logged this, or involved internal user
  type: communicationTypeEnum("type").notNull(),
  direction: communicationDirectionEnum("direction").notNull(),
  timestamp: timestamp("timestamp").notNull().defaultNow(), // When the actual communication happened
  source: text("source"), // e.g., 'gmail_sync', 'twilio_sync', 'manual_entry', 'system_generated'
  externalId: text("external_id"), // Unique ID from the external system (e.g., email Message-ID, call SID)
  subject: text("subject"), // For emails or meeting titles
  fromAddress: text("from_address"), // For emails
  toAddress: text("to_address"), // For emails (could be an array, consider how to store if multiple)
  bodyRaw: text("body_raw"), // Full email body, call transcript, etc.
  bodySummary: text("body_summary"), // AI-generated summary or user-provided summary
  durationMinutes: integer("duration_minutes"), // For calls
  recordingUrl: text("recording_url"), // For call recordings
  metaData: jsonb("meta_data"), // Any other structured data (e.g., email headers, call tags, AI analysis results)
  createdAt: timestamp("created_at").defaultNow().notNull(),
  updatedAt: timestamp("updated_at").defaultNow().notNull(),
});

Key Schema Changes and Explanations:

contactIdentifiers Table:

leadId & clientId: Foreign keys to link the identifier to either a lead or a client. Using onDelete: 'cascade' means if a lead/client is deleted, their associated contact identifiers are also removed.
type: An enum (identifierTypeEnum) to specify if it's an 'email' or 'phone'.
value: The actual email address or phone number.
isPrimary: A boolean to mark one email/phone as the primary contact method.
source: Text field to note how this contact identifier was captured.
communications Table:

leadId & clientId: Foreign keys to link the communication. onDelete: 'set null' means if a lead/client is deleted, the communication record remains but is no longer directly linked (you might choose 'cascade' if you want to delete communications too).
userId: Foreign key to users.id. This can represent the internal user who logged the communication or was involved.
type: Enum (communicationTypeEnum) for 'email', 'call', 'sms', 'note', 'meeting'.
direction: Enum (communicationDirectionEnum) for 'incoming', 'outgoing', or 'internal' (for notes).
timestamp: When the communication occurred (important for timeline sorting).
source: Where this communication record originated (e.g., 'gmail_sync', 'manual_entry').
externalId: For storing unique IDs from external systems (like an email's Message-ID or a call's SID from Twilio) to help with deduplication.
subject, fromAddress, toAddress: Specific to emails.
bodyRaw: To store the full content. Be mindful of potential size.
bodySummary: For AI-generated or manual summaries.
durationMinutes, recordingUrl: Specific to calls.
metaData: A jsonb field for any other structured information specific to the communication type or source.
Enum Definitions (pgEnum):

Added identifierTypeEnum and communicationTypeEnum, communicationDirectionEnum for PostgreSQL enum types, which Drizzle ORM supports.
Zod Schemas and Types:

Added insertContactIdentifierSchema, insertCommunicationSchema and their corresponding TypeScript types (ContactIdentifier, Communication, InsertContactIdentifier, InsertCommunication).
